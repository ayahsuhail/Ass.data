def find_repeated_element(B):
    distinct_elements = set()  # مجموعة لتخزين العناصر التي تمت مشاهدتها

    for b in B:  # التكرار عبر العناصر
        if b in distinct_elements:  # إذا كان العنصر موجودًا بالفعل في المجموعة
            return b  # إرجاع العنصر المكرر
        distinct_elements.add(b)  # إضافة العنصر إلى المجموعة إذا لم يكن مكررًا

    # في حال لم يوجد عنصر مكرر (لكن من المفترض أن يوجد بناءً على شروط المسألة)
    return None

تعقيد الزمن:

تعقيد الزمن: O(n)، حيث n هو حجم المصفوفة B. نقوم بالتكرار عبر المصفوفة مرة واحدة، وكل عملية بحث وإضافة في المجموعة تتم في وقت ثابت O(1) في المتوسط.

تعقيد المساحة: O(n)، لأننا نحتاج لتخزين جميع العناصر في المجموعة.


صحة الخوارزمية:

بما أن المصفوفة تحتوي على n عنصرًا والقيم تتراوح بين 1 و n-1 مع وجود عنصر مكرر واحد فقط، فإن هذه الخوارزمية تعمل بشكل صحيح لاكتشاف العنصر المكرر باستخدام المجموعة.
