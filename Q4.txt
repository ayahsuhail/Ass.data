 تنفيذ طريقة clone() لمكدس يستخدم المصفوفات.

الحل:

public class ArrayStack implements Cloneable {
    private int[] stack;
    private int top;

    public ArrayStack(int capacity) {
        stack = new int[capacity];
        top = -1;
    }

    public void push(int value) {
        if (top == stack.length - 1) throw new StackOverflowError();
        stack[++top] = value;
    }

    public int pop() {
        if (top == -1) throw new IllegalStateException("Stack is empty");
        return stack[top--];
    }

    public boolean isEmpty() {
        return top == -1;
    }

    @Override
    protected ArrayStack clone() {
        ArrayStack clonedStack = new ArrayStack(stack.length);
        clonedStack.top = this.top;
        System.arraycopy(this.stack, 0, clonedStack.stack, 0, this.top + 1);
        return clonedStack;
    }

    public static void main(String[] args) {
        ArrayStack original = new ArrayStack(5);
        original.push(10);
        original.push(20);

        ArrayStack cloned = original.clone();
        System.out.println("Original stack top: " + original.pop());
        System.out.println("Cloned stack top: " + cloned.pop());
    }
}
