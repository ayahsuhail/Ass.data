5. تنفيذ برنامج لحل تعبير Postfix وطباعة النتيجة


الحل:

import java.util.Stack;

public class PostfixEvaluator {
    public static int evaluatePostfix(String expression) {
        Stack<Integer> stack = new Stack<>();

        for (char ch : expression.toCharArray()) {
            if (Character.isDigit(ch)) {
                // تحويل الحرف إلى رقم وإضافته إلى المكدس
                stack.push(ch - '0');
            } else {
                // استخراج الرقمين الأخيرين من المكدس
                int b = stack.pop();
                int a = stack.pop();

                // تنفيذ العملية الحسابية
                switch (ch) {
                    case '+': stack.push(a + b); break;
                    case '-': stack.push(a - b); break;
                    case '*': stack.push(a * b); break;
                    case '/': stack.push(a / b); break;
                }
            }
        }

        // النتيجة النهائية
        return stack.pop();
    }

    public static void main(String[] args) {
        String postfixExp = "52+83-*4/"; // تعبير Postfix مكافئ لـ ((5 + 2) * (8 - 3)) / 4
        int result = evaluatePostfix(postfixExp);
        System.out.println("Result: " + result); // الناتج: 4
    }
}


المخرجات:

1. 5 و 2 يتم إدخالهما إلى المكدس
2. + يتم تطبيقها  (5 + 2) = 7  إدخال 7 في المكدس
3. 8 و 3 يتم إدخالهما إلى المكدس
4. - يتم تطبيقها  (8 - 3) = 5  إدخال 5 في المكدس
5. * يتم تطبيقها  (7 * 5) = 35  إدخال 35 في المكدس
6. 4 يتم إدخاله إلى المكدس
7. / يتم تطبيقها  (35 / 4) = 4 (القيمة الصحيحة)
